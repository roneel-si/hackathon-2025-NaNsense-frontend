                        <div class="progress-fill" id="progressFill"></div>
                    </div>
                </div>
            </div>
        </div>
    </div>

    <script>
        const quizData = [
            {
                question: "In which year was Rajasthan Royals founded?",
                options: ["2007", "2008", "2009", "2006"],
                correct: 0,
                alternateQuestion: "What is the team motto of Rajasthan Royals?",
                alternateOptions: ["Halla Bol", "Korbo Lorbo Jeetbo", "Whistle Podu", "Play Bold"],
                alternateCorrect: 0
            },
            {
                question: "Who was the captain when RR won their first IPL title?",
                options: ["Rahul Dravid", "Shane Warne", "Ajinkya Rahane", "Steve Smith"],
                correct: 1,
                alternateQuestion: "Which legendary spinner mentored RR in their early years?",
                alternateOptions: ["Anil Kumble", "Shane Warne", "Harbhajan Singh", "Daniel Vettori"],
                alternateCorrect: 1
            },
            {
                question: "What is the home ground of Rajasthan Royals?",
                options: ["Sawai Mansingh Stadium", "Sardar Patel Stadium", "M. Chinnaswamy Stadium", "Eden Gardens"],
                correct: 0,
                alternateQuestion: "In which city is RR's home ground located?",
                alternateOptions: ["Jodhpur", "Udaipur", "Jaipur", "Kota"],
                alternateCorrect: 2
            },
            {
                question: "Which color is predominantly associated with RR's jersey?",
                options: ["Blue", "Pink", "Yellow", "Red"],
                correct: 1,
                alternateQuestion: "What is the secondary color in RR's team colors?",
                alternateOptions: ["Silver", "Gold", "White", "Blue"],
                alternateCorrect: 1
            },
            {
                question: "Who is the current wicket-keeper batsman for RR?",
                options: ["MS Dhoni", "Jos Buttler", "Dinesh Karthik", "Sanju Samson"],
                correct: 3,
                alternateQuestion: "Which overseas player is known as RR's 'Mr. 360'?",
                alternateOptions: ["Ben Stokes", "Jos Buttler", "Chris Morris", "Liam Livingstone"],
                alternateCorrect: 1
            }
        ];

        let currentQuestion = 0;
        let score = 0;
        let selectedOption = -1;
        let quizCompleted = false;
        let timeLeft = 15;
        let timerInterval;
        let isAnswered = false;
        
        // Lifelines
        let fiftyFiftyUsed = false;
        let questionFlipUsed = false;
        let extendTimeUsed = false;

        function openQuiz() {
            document.getElementById('quizOverlay').classList.add('active');
            showStartScreen();
        }

        function closeQuiz() {
            document.getElementById('quizOverlay').classList.remove('active');
            clearInterval(timerInterval);
        }

        function showStartScreen() {
            document.getElementById('startScreen').classList.remove('hidden');
            document.getElementById('quizGame').classList.add('hidden');
        }

        function startQuiz() {
            document.getElementById('startScreen').classList.add('hidden');
            document.getElementById('quizGame').classList.remove('hidden');
            resetQuiz();
        }

        function resetQuiz() {
            currentQuestion = 0;
            score = 0;
            selectedOption = -1;
            quizCompleted = false;
            fiftyFiftyUsed = false;
            questionFlipUsed = false;
            extendTimeUsed = false;
            
            // Reset lifelines
            document.getElementById('fiftyFifty').classList.remove('used');
            document.getElementById('questionFlip').classList.remove('used');
            document.getElementById('extendTime').classList.remove('used');
            
            // Show timer and lifelines
            document.querySelector('.timer-container').style.display = 'flex';
            document.querySelector('.lifelines').style.display = 'flex';
            
            showQuestion();
        }

        function showQuestion() {
            if (currentQuestion >= quizData.length) {
                showResults();
                return;
            }

            isAnswered = false;
            timeLeft = 15;
            clearInterval(timerInterval);

            const question = quizData[currentQuestion];
            const questionCounter = document.getElementById('questionCounter');
            const questionText = document.getElementById('questionText');
            const optionsContainer = document.getElementById('optionsContainer');
            const progressFill = document.getElementById('progressFill');

            // Update question counter and progress
            questionCounter.textContent = `Question ${currentQuestion + 1} of ${quizData.length}`;
            progressFill.style.width = `${((currentQuestion) / quizData.length) * 100}%`;

            // Update question text
            questionText.textContent = question.question;

            // Clear and populate options
            optionsContainer.innerHTML = '';
            selectedOption = -1;

            question.options.forEach((option, index) => {
                const optionDiv = document.createElement('div');
                optionDiv.className = 'option';
                optionDiv.textContent = option;
                optionDiv.onclick = () => selectOption(index);
                optionsContainer.appendChild(optionDiv);
            });

            // Animate options
            setTimeout(() => {
                document.querySelectorAll('.option').forEach((opt, index) => {
                    setTimeout(() => {
                        opt.classList.add('animate');
                    }, index * 100);
                });
            }, 100);

            // Start timer
            startTimer();
        }

        function startTimer() {
            const timerText = document.getElementById('timerText');
            const timerCircle = document.getElementById('timerCircle');
            const timer = document.getElementById('timer');

            timerInterval = setInterval(() => {
                timeLeft--;
                timerText.textContent = timeLeft;
                
                // Update circular progress
                const percentage = ((15 - timeLeft) / 15) * 360;
                timerCircle.style.background = `conic-gradient(#ffd700 ${percentage}deg, transparent ${percentage}deg)`;
                
                // Warning state
                if (timeLeft <= 5) {
                    timer.classList.add('warning');
                    timerCircle.style.background = `conic-gradient(#ff4444 ${percentage}deg, transparent ${percentage}deg)`;
                } else {
                    timer.classList.remove('warning');
                }
                
                // Time's up
                if (timeLeft <= 0) {
                    clearInterval(timerInterval);
                    if (!isAnswered) {
                        timeUp();
                    }
                }
            }, 1000);
        }

        function timeUp() {
            isAnswered = true;
            const question = quizData[currentQuestion];
            const options = document.querySelectorAll('.option');
            
            options[question.correct].classList.add('correct');
            options.forEach(opt => opt.classList.add('disabled'));
            
            setTimeout(() => {
                currentQuestion++;
                showQuestion();
            }, 2000);
        }

        function selectOption(index) {
            if (isAnswered) return;
            
            isAnswered = true;
            clearInterval(timerInterval);
            
            const question = quizData[currentQuestion];
            const options = document.querySelectorAll('.option');
            
            // Mark selected option
            options[index].classList.add('selected');
            
            // Show correct/incorrect feedback
            options[question.correct].classList.add('correct');
            if (index !== question.correct) {
                options[index].classList.add('incorrect');
            } else {
                score++;
            }
            
            // Disable all options
            options.forEach(opt => opt.classList.add('disabled'));
            
            // Move to next question after delay
            setTimeout(() => {
                currentQuestion++;
                showQuestion();
            }, 2000);
        }

        // Lifeline Functions
        function useFiftyFifty() {
            if (fiftyFiftyUsed || isAnswered) return;
            
            fiftyFiftyUsed = true;
            document.getElementById('fiftyFifty').classList.add('used');
            
            const question = quizData[currentQuestion];
            const options = document.querySelectorAll('.option');
            const correctIndex = question.correct;
            
            let wrongOptionsHidden = 0;
            options.forEach((option, index) => {
                if (index !== correctIndex && wrongOptionsHidden < 2) {
                    option.classList.add('hidden');
                    wrongOptionsHidden++;
                }
            });
        }

        function useQuestionFlip() {
            if (questionFlipUsed || isAnswered) return;
            
            questionFlipUsed = true;
            document.getElementById('questionFlip').classList.add('used');
            
            const question = quizData[currentQuestion];
            const questionText = document.getElementById('questionText');
            const optionsContainer = document.getElementById('optionsContainer');
            
            // Clear timer
            clearInterval(timerInterval);
            
            // Update to alternate question
            questionText.textContent = question.alternateQuestion;
            optionsContainer.innerHTML = '';
            
            question.alternateOptions.forEach((option, index) => {
                const optionDiv = document.createElement('div');
                optionDiv.className = 'option';
                optionDiv.textContent = option;
                optionDiv.onclick = () => selectAlternateOption(index);
                optionsContainer.appendChild(optionDiv);
            });
            
            // Animate new options
            setTimeout(() => {
                document.querySelectorAll('.option').forEach((opt, index) => {
                    setTimeout(() => {
                        opt.classList.add('animate');
                    }, index * 100);
                });
            }, 100);
            
            // Reset timer
            timeLeft = 15;
            isAnswered = false;
            startTimer();
        }

        function selectAlternateOption(index) {
            if (isAnswered) return;
            
            isAnswered = true;
            clearInterval(timerInterval);
            
            const question = quizData[currentQuestion];
            const options = document.querySelectorAll('.option');
            
            options[index].classList.add('selected');
            options[question.alternateCorrect].classList.add('correct');
            
            if (index !== question.alternateCorrect) {
                options[index].classList.add('incorrect');
            } else {
                score++;
            }
            
            options.forEach(opt => opt.classList.add('disabled'));
            
            setTimeout(() => {
                currentQuestion++;
                showQuestion();
            }, 2000);
        }

        function useExtendTime() {
            if (extendTimeUsed || isAnswered) return;
            
            extendTimeUsed = true;
            document.getElementById('extendTime').classList.add('used');
            
            timeLeft += 10;
            if (timeLeft > 25) timeLeft = 25; // Cap at 25 seconds
        }

        function showResults() {
            clearInterval(timerInterval);
            const percentage = Math.round((score / quizData.length) * 100);
            let message = '';
            
            if (percentage >= 80) {
                message = "üèÜ Outstanding! You're a true RR superfan!";
                // Trigger confetti for high scores
                createConfetti();
            } else if (percentage >= 60) {
                message = "üëè Great job! You know your Royals well!";
            } else if (percentage >= 40) {
                message = "üëç Not bad! Keep following the Royals!";
            } else {
                message = "üìö Time to brush up on your RR knowledge!";
            }

            const quizContent = document.getElementById('quizContent');
            quizContent.innerHTML = `
                <div class="quiz-results">
                    <div class="score-display">${score}/${quizData.length}</div>
                    <div class="score-message">${message}</div>
                    <button class="restart-btn" onclick="startQuiz()">Take Quiz Again</button>
                </div>
            `;

            // Hide timer and lifelines
            document.querySelector('.timer-container').style.display = 'none';
            document.querySelector('.lifelines').style.display = 'none';
            
            // Update progress to 100%
            document.getElementById('progressFill').style.width = '100%';
        }

        function createConfetti() {
            // Create confetti container
            const confettiContainer = document.createElement('div');
            confettiContainer.className = 'confetti-container';
            document.body.appendChild(confettiContainer);

            // Create multiple confetti pieces
            for (let i = 0; i < 50; i++) {
                const confetti = document.createElement('div');
                confetti.className = 'confetti';
                
                // Random positioning and timing
                confetti.style.left = Math.random() * 100 + '%';
                confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';
                confetti.style.animationDelay = Math.random() * 2 + 's';
                
                // Random shapes (some rectangular, some square)
                if (Math.random() > 0.5) {
                    confetti.style.width = '12px';
                    confetti.style.height = '6px';
                } else {
                    confetti.style.width = '8px';
                    confetti.style.height = '8px';
                    confetti.style.borderRadius = '50%';
                }
                
                confettiContainer.appendChild(confetti);
            }

            // Add burst effect to score display
            const scoreDisplay = document.querySelector('.score-display');
            if (scoreDisplay) {
                scoreDisplay.classList.add('confetti-burst');
            }

            // Remove confetti after animation completes
            setTimeout(() => {
                if (confettiContainer.parentNode) {
                    confettiContainer.parentNode.removeChild(confettiContainer);
                }
            }, 5000);
        }

        // Close quiz when clicking outside
        document.getElementById('quizOverlay').addEventListener('click', function(e) {
            if (e.target === this) {
                closeQuiz();
            }
        });
    </script>
